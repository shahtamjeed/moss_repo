Count = namedtuple('Count', 'letter number')

def letter_count (string: str, key_str: str) -> list:
    '''The function returns the counts of certain letters in the first
    parameter; the second parameter is a string that specifies which letters
    to count.'''
    count_list = []
    for i in range(len(key_str)):
        count = 0
        for j in range(len(string)):
            if string[j] == key_str[i]:
                count = count + 1
        count_list.append(Count(key_str[i], count))
    return count_list

assert letter_count('The cabbage has baggage', 'abcd') == [Count(letter='a', number=5),
                                                           Count(letter='b', number=3),
                                                           Count(letter='c', number=1),
                                                           Count(letter='d', number=0)]
print (letter_count('The cabbage has baggage', 'abcd'))
